    <script>
        // Global variables
        let currentSessionId = null;
        const username = "{{ session.username }}";
        
        // DOM Elements
        const usernameDisplay = document.getElementById('username-display');
        const newChatBtn = document.getElementById('new-chat-btn');
        const chatSessionsList = document.getElementById('chat-sessions');
        const messagesContainer = document.getElementById('messages-container');
        const messageForm = document.getElementById('message-form');
        const messageInput = document.getElementById('message-input');
        const currentSessionTitle = document.getElementById('current-session-title');
        const exportChatBtn = document.getElementById('export-chat');
        const deleteChatBtn = document.getElementById('delete-chat');
        
        // Set username
        if (username) {
            usernameDisplay.textContent = username;
        }
        
        // Load chat sessions
        function loadChatSessions() {
            fetch('/sessions')
                .then(response => response.json())
                .then(sessions => {
                    chatSessionsList.innerHTML = '';
                    sessions.forEach(session => {
                        const li = document.createElement('li');
                        li.innerHTML = `<i class="fas fa-comment"></i> ${session.name}`;
                        li.dataset.sessionId = session.id;
                        
                        if (session.id === currentSessionId) {
                            li.classList.add('active');
                        }
                        
                        li.addEventListener('click', () => loadChatSession(session.id));
                        chatSessionsList.appendChild(li);
                    });
                });
        }
        
        // Load a specific chat session
        function loadChatSession(sessionId) {
            currentSessionId = sessionId;
            currentSessionTitle.textContent = `Chat Session ${sessionId}`;
            
            // Highlight active session in sidebar
            document.querySelectorAll('#chat-sessions li').forEach(li => {
                li.classList.remove('active');
                if (li.dataset.sessionId === sessionId.toString()) {
                    li.classList.add('active');
                }
            });
            
            // Load messages
            fetch(`/history/${sessionId}`)
                .then(response => response.json())
                .then(messages => {
                    messagesContainer.innerHTML = '';
                    
                    if (messages.length === 0) {
                        messagesContainer.innerHTML = `
                            <div class="welcome-message">
                                <h2>New Chat Session</h2>
                                <p>Start chatting with the AI assistant.</p>
                            </div>
                        `;
                        return;
                    }
                    
                    messages.forEach(msg => {
                        addMessageToUI(msg.type, msg.text, msg.timestamp);
                    });
                    
                    // Scroll to bottom
                    messagesContainer.scrollTop = messagesContainer.scrollHeight;
                });
        }
        
        // Create new chat session
        function createNewChatSession() {
            fetch('/chat', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify({
                    message: '',
                    session_id: null
                })
            })
            .then(response => response.json())
            .then(data => {
                currentSessionId = data.session_id;
                currentSessionTitle.textContent = `New Chat`;
                messagesContainer.innerHTML = `
                    <div class="welcome-message">
                        <h2>New Chat Session</h2>
                        <p>Start chatting with the AI assistant.</p>
                    </div>
                `;
                loadChatSessions();
            });
        }
        
        // Add message to UI
        function addMessageToUI(type, text, timestamp) {
            const messageDiv = document.createElement('div');
            messageDiv.className = `message message-${type}`;
            
            const avatarDiv = document.createElement('div');
            avatarDiv.className = 'message-avatar';
            avatarDiv.textContent = type === 'user' ? username.charAt(0).toUpperCase() : 'AI';
            
            const contentDiv = document.createElement('div');
            contentDiv.className = 'message-content';
            contentDiv.innerHTML = text.replace(/\n/g, '<br>');
            
            const timeDiv = document.createElement('div');
            timeDiv.className = 'message-time';
            timeDiv.textContent = new Date(timestamp).toLocaleTimeString();
            
            if (type === 'user') {
                messageDiv.appendChild(contentDiv);
                messageDiv.appendChild(avatarDiv);
                messageDiv.appendChild(timeDiv);
            } else {
                messageDiv.appendChild(avatarDiv);
                messageDiv.appendChild(contentDiv);
                messageDiv.appendChild(timeDiv);
            }
            
            messagesContainer.appendChild(messageDiv);
            
            // Scroll to bottom
            messagesContainer.scrollTop = messagesContainer.scrollHeight;
        }
        
        // Handle form submission
        messageForm.addEventListener('submit', function(e) {
            e.preventDefault();
            const message = messageInput.value.trim();
            
            if (!message) return;
            
            // Add user message to UI immediately
            addMessageToUI('user', message, new Date().toISOString());
            
            // Clear input
            messageInput.value = '';
            
            // Send to server
            fetch('/chat', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify({
                    message: message,
                    session_id: currentSessionId || null
                })
            })
            .then(response => response.json())
            .then(data => {
                if (data.error) {
                    addMessageToUI('bot', "Sorry, I encountered an error. Please try again.", new Date().toISOString());
                    console.error(data.error);
                    return;
                }
                
                // Update current session if this was a new chat
                if (!currentSessionId) {
                    currentSessionId = data.session_id;
                    currentSessionTitle.textContent = `Chat Session ${currentSessionId}`;
                    loadChatSessions();
                }
                
                // Add bot response
                addMessageToUI('bot', data.response, new Date().toISOString());
            })
            .catch(error => {
                addMessageToUI('bot', "Sorry, I'm having trouble connecting to the server.", new Date().toISOString());
                console.error('Error:', error);
            });
        });
        
        // New chat button
        newChatBtn.addEventListener('click', createNewChatSession);
        
        // Export chat
        exportChatBtn.addEventListener('click', () => {
            if (currentSessionId) {
                window.open(`/export/${currentSessionId}`, '_blank');
            }
        });
        
        // Delete chat
        deleteChatBtn.addEventListener('click', () => {
            if (currentSessionId && confirm('Are you sure you want to delete this chat session?')) {
                // Implement delete functionality in your Flask backend
                fetch(`/sessions/${currentSessionId}`, {
                    method: 'DELETE'
                })
                .then(response => {
                    if (response.ok) {
                        createNewChatSession();
                        loadChatSessions();
                    }
                });
            }
        });
        
        // Initial load
        loadChatSessions();
    </script>
